205torch::Tensor NeuralNetworkWrapper::forward(DataType data_type) {
206 if (!is_initialized_) {
207 throw std::runtime_error("Network must be initialized before forward pass. Call initializeNetwork() first.");
208 }
209
210 if (layers_->size() == 0) {
211 throw std::runtime_error("No layers defined in the network.");
212 }
213
214 // Get the appropriate input data based on data type
215 if (!hasInputData(data_type)) {
216 std::string data_type_str = (data_type == DataType::Train) ? "training" : "test";
217 throw std::runtime_error("No " + data_type_str + " input data available. Use setInputData() first.");
218 }
219
220 torch::Tensor input = getInputData(data_type);
221
222 // Validate input dimensions
223 if (input.dim() != 2) {
224 throw std::runtime_error("Input must be a 2D tensor with shape [batch_size, features].");
225 }
226
227 if (input.size(1) != input_size_) {
228 throw std::runtime_error("Input feature size (" + std::to_string(input.size(1)) +
229 ") doesn't match network input size (" + std::to_string(input_size_) + ").");
230 }
231
232 torch::Tensor x = input;
233
234 // Forward pass through all layers except the last one (with activation)
235 for (size_t i = 0; i < layers_->size() - 1; ++i) {
236 // Apply linear transformation
237 x = layers_[i]->as<torch::nn::Linear>()->forward(x);
238
239 // Apply activation function
240 if (activation_function_ == "relu") {
241 x = torch::relu(x);
242 } else if (activation_function_ == "tanh") {
243 x = torch::tanh(x);
244 } else if (activation_function_ == "sigmoid") {
245 x = torch::sigmoid(x);
246 } else {
247 throw std::runtime_error("Unknown activation function: " + activation_function_);
248 }
249 }
250
251 // Final layer without activation (for regression)
252 if (layers_->size() > 0) {
253 x = layers_[layers_->size() - 1]->as<torch::nn::Linear>()->forward(x);
254 }
255
256 return x;
257}
258std::vector<double> NeuralNetworkWrapper::train(int num_epochs,
259 int batch_size,
260 double learning_rate) {
261 if (!is_initialized_) {
262 throw std::runtime_error("Network must be initialized before training. Call initializeNetwork() first.");
263 }
264
265 if (!hasInputData(DataType::Train) || !hasTargetData(DataType::Train)) {
266 throw std::runtime_error("Training data not available. Use setInputData() and setOutputData() with DataType::Train first.");
267 }
268
269 // Get training data
270 torch::Tensor train_inputs = getInputData(DataType::Train);
271 torch::Tensor train_targets = getTargetData(DataType::Train);
272
273 // Validate data compatibility
274 if (train_inputs.size(0) != train_targets.size(0)) {
275 throw std::runtime_error("Training input and output data must have the same number of samples.");
276 }
277
278 const int num_train_samples = train_inputs.size(0);
279
280 // Create optimizer
281 torch::optim::Adam optimizer(this->parameters(), torch::optim::AdamOptions(learning_rate));
282
283 // Clear previous training history
284 training_history_.clear();
285 training_history_.reserve(num_epochs);
286
287 std::cout << "Starting training..." << std::endl;
288 std::cout << "Epochs: " << num_epochs << ", Batch size: " << batch_size
289 << ", Learning rate: " << learning_rate << std::endl;
290 std::cout << "Training samples: " << num_train_samples << std::endl;
291
292 // Training loop
293 for (int epoch = 0; epoch < num_epochs; ++epoch) {
294 double total_loss = 0.0;
295 int num_batches = 0;
296
297 // Mini-batch training
298 for (int i = 0; i < num_train_samples; i += batch_size) {
299 int current_batch_size = std::min(batch_size, num_train_samples - i);
300
301 // Get batch
302 torch::Tensor batch_inputs = train_inputs.slice(0, i, i + current_batch_size);
303 torch::Tensor batch_targets = train_targets.slice(0, i, i + current_batch_size);
304
305 // Zero gradients
306 optimizer.zero_grad();
307
308 // Forward pass
309 torch::Tensor predictions = this->forward_internal(batch_inputs);
310
311 // Compute loss
312 torch::Tensor loss = torch::mse_loss(predictions, batch_targets);
313
314 // Backward pass
315 loss.backward();
316
317 // Update weights
318 optimizer.step();
319
320 total_loss += loss.item<double>();
321 num_batches++;
322 }
323
324 double avg_loss = total_loss / num_batches;
325 current_loss_ = avg_loss;
326 training_history_.push_back(avg_loss);
327
328 // Print progress
329 if ((epoch + 1) % 20 == 0 || epoch == 0) {
330 std::cout << "Epoch [" << (epoch + 1) << "/" << num_epochs
331 << "], Average Loss: " << std::fixed << std::setprecision(6)
332 << avg_loss << std::endl;
333 }
334 }
335
336 std::cout << "Training completed!" << std::endl;
337 return training_history_;
338}

460std::map<std::string, double> NeuralNetworkWrapper::computeMetrics(const torch::Tensor& predictions,
461 const torch::Tensor& targets) {
462 // TODO: Implement metrics computation

464 std::map<std::string, double> metrics;
465
466 // Mean Squared Error
467 torch::Tensor mse = torch::mse_loss(predictions, targets);
468 metrics["mse"] = mse.item<double>();
469
470 // Root Mean Squared Error
471 metrics["rmse"] = std::sqrt(metrics["mse"]);
472
473 // Mean Absolute Error
474 torch::Tensor mae = torch::mean(torch::abs(predictions - targets));
475 metrics["mae"] = mae.item<double>();
476
477 // R-squared
478 torch::Tensor target_mean = torch::mean(targets);
479 torch::Tensor ss_res = torch::sum(torch::pow(targets - predictions, 2));
480 torch::Tensor ss_tot = torch::sum(torch::pow(targets - target_mean, 2));
481 double r_squared = 1.0 - (ss_res.item<double>() / ss_tot.item<double>());
482 metrics["r_squared"] = r_squared;
483
484 return metrics;

487void NeuralNetworkWrapper::setInputData(DataType data_type, const TimeSeriesSet<double>& time_series_set,
488 double t_start, double t_end, double dt) {
489 if (lags_.empty()) {
490 throw std::runtime_error("No lag configuration set. Use setLags() before setting input data.");
491 }
492
493 setOriginalSeriesNames(time_series_set.getSeriesNames());
494
495 // Calculate total number of features based on lags
496 int total_features = 0;
497 for (const auto& series_lags : lags_) {
498 total_features += series_lags.size();
499 }
500
501 if (total_features == 0) {
502 throw std::runtime_error("No lag features configured. Check lag configuration.");
503 }
504
505 // Calculate number of time steps
506 int num_time_steps = static_cast<int>((t_end - t_start) / dt) + 1;
507
508 // Create output tensor [num_samples, total_features]
509 torch::Tensor input_tensor = torch::zeros({num_time_steps, total_features}, torch::kFloat32);
510
511 int feature_index = 0;
512
513 // Process each TimeSeries with its corresponding lags
514 for (size_t series_idx = 0; series_idx < lags_.size(); ++series_idx) {
515 if (series_idx >= time_series_set.size()) {
516 throw std::runtime_error("Lag configuration references TimeSeries index " +
517 std::to_string(series_idx) + " but TimeSeriesSet only has " +
518 std::to_string(time_series_set.size()) + " series.");
519 }
520
521 const auto& current_series = time_series_set[series_idx];
522 const auto& series_lags = lags_[series_idx];
523
524 // For each lag in this series
525 for (int lag : series_lags) {
526 // Fill the feature column for this lag
527 for (int time_idx = 0; time_idx < num_time_steps; ++time_idx) {
528 double current_time = t_start + time_idx * dt;
529 double lag_time = current_time - lag * dt; // Apply lag
530
531 // Interpolate value at lagged time
532 double lagged_value = current_series.interpol(lag_time);
533 input_tensor[time_idx][feature_index] = static_cast<float>(lagged_value);
534 }
535 feature_index++;
536 }
537 }
538
539 // Store the processed input data based on data type
540 if (data_type == DataType::Train) {
541 train_input_data_ = input_tensor;
542 } else {
543 test_input_data_ = input_tensor;
544 }
545
546 // Validate input dimensions if network is initialized
547 if (is_initialized_ && total_features != input_size_) {
548 throw std::runtime_error("Generated input features (" + std::to_string(total_features) +
549 ") don't match network input size (" + std::to_string(input_size_) +
550 "). Network may need to be reinitialized.");
551 }
552}

554void NeuralNetworkWrapper::setTargetData(DataType data_type, const TimeSeriesSet<double>& time_series_set,
555 const std::vector<int>& output_indices,
556 double t_start, double t_end, double dt) {
557 if (output_indices.empty()) {
558 throw std::runtime_error("No output indices specified. Provide at least one TimeSeries index.");
559 }
560
561 // Validate all indices are within bounds
562 for (int idx : output_indices) {
563 if (idx < 0 || idx >= static_cast<int>(time_series_set.size())) {
564 throw std::runtime_error("Output index " + std::to_string(idx) +
565 " is out of range. TimeSeriesSet has " +
566 std::to_string(time_series_set.size()) + " series.");
567 }
568 }
569
570 // Calculate number of time steps and output features
571 int num_time_steps = static_cast<int>((t_end - t_start) / dt) + 1;
572 int num_outputs = output_indices.size();
573
574 // Create output tensor [num_samples, num_outputs]
575 torch::Tensor output_tensor = torch::zeros({num_time_steps, num_outputs}, torch::kFloat32);
576
577 // Fill tensor with interpolated values from selected TimeSeries
578 for (int time_idx = 0; time_idx < num_time_steps; ++time_idx) {
579 double current_time = t_start + time_idx * dt;
580
581 for (int output_idx = 0; output_idx < num_outputs; ++output_idx) {
582 int series_index = output_indices[output_idx];
583 const auto& series = time_series_set[series_index];
584
585 double interpolated_value = series.interpol(current_time);
586 output_tensor[time_idx][output_idx] = static_cast<float>(interpolated_value);
587 }
588 }
589
590 // Store the processed output data based on data type
591 if (data_type == DataType::Train) {
592 train_target_data_ = output_tensor;
593 } else {
594 test_target_data_ = output_tensor;
595 }
596
597 // Validate output dimensions if network is initialized
598 if (is_initialized_ && num_outputs != output_size_) {
599 throw std::runtime_error("Generated output features (" + std::to_string(num_outputs) +
600 ") don't match network output size (" + std::to_string(output_size_) +
601 "). Network may need to be reinitialized.");
602 }
603
604 // Validate sample count compatibility with input data
605 if (hasInputData(data_type)) {
606 torch::Tensor input_data = getInputData(data_type);
607 if (num_time_steps != input_data.size(0)) {
608 throw std::runtime_error("Output data sample count (" + std::to_string(num_time_steps) +
609 ") doesn't match input data sample count (" + std::to_string(input_data.size(0)) + ")");
610 }
611 }

697torch::Tensor NeuralNetworkWrapper::forward_internal(torch::Tensor input) {
698 torch::Tensor x = input;
699
700 // Forward pass through all layers except the last one (with activation)
701 for (size_t i = 0; i < layers_->size() - 1; ++i) {
702 x = layers_[i]->as<torch::nn::Linear>()->forward(x);
703
704 // Apply activation function
705 if (activation_function_ == "relu") {
706 x = torch::relu(x);
707 } else if (activation_function_ == "tanh") {
708 x = torch::tanh(x);
709 } else if (activation_function_ == "sigmoid") {
710 x = torch::sigmoid(x);
711 }
712 }
713
714 // Final layer without activation
715 if (layers_->size() > 0) {
716 x = layers_[layers_->size() - 1]->as<torch::nn::Linear>()->forward(x);
717 }
718
719 return x;
720}

722TimeSeriesSet<double> NeuralNetworkWrapper::predict(DataType data_type,
723 double t_start, double t_end, double dt,
724 const std::vector<std::string>& output_names) {
725 if (!is_initialized_) {
726 throw std::runtime_error("Network must be initialized before prediction. Call initializeNetwork() first.");
727 }
728
729 if (!hasInputData(data_type)) {
730 std::string data_type_str = (data_type == DataType::Train) ? "training" : "test";
731 throw std::runtime_error("No " + data_type_str + " input data available. Use setInputData() first.");
732 }
733
734 // Get predictions from the network
735 torch::Tensor predictions;
736 {
737 torch::NoGradGuard no_grad; // Disable gradient computation for inference
738 predictions = forward(data_type);
739 }
740
741 // Calculate number of time steps
742 int num_time_steps = static_cast<int>((t_end - t_start) / dt) + 1;
743
744 // Validate that predictions match expected time steps
745 if (predictions.size(0) != num_time_steps) {
746 throw std::runtime_error("Prediction tensor size (" + std::to_string(predictions.size(0)) +
747 ") doesn't match expected time steps (" + std::to_string(num_time_steps) + ")");
748 }
749
750 // Get number of output features
751 int num_outputs = predictions.size(1);
752
753 // Create TimeSeriesSet for results
754 TimeSeriesSet<double> result(num_outputs);
755
756 // Set names for each output TimeSeries
757 for (int output_idx = 0; output_idx < num_outputs; ++output_idx) {
758 std::string name;
759 if (output_idx < static_cast<int>(output_names.size()) && !output_names[output_idx].empty()) {
760 name = output_names[output_idx];
761 } else {
762 name = "output_" + std::to_string(output_idx);
763 }
764 result.setSeriesName(output_idx, name);
765 }
766
767 // Fill TimeSeries with predicted values
768 for (int time_idx = 0; time_idx < num_time_steps; ++time_idx) {
769 double current_time = t_start + time_idx * dt;
770 for (int output_idx = 0; output_idx < num_outputs; ++output_idx) {
771 double predicted_value = predictions[time_idx][output_idx].item<double>();
772 result[output_idx].addPoint(current_time, predicted_value);
773 }
774 }
775
776 return result;
777}

779double NeuralNetworkWrapper::calculateR2(DataType data_type) {
780 if (!is_initialized_) {
781 throw std::runtime_error("Network must be initialized before calculating R². Call initializeNetwork() first.");
782 }
783
784 if (!hasInputData(data_type) || !hasTargetData(data_type)) {
785 std::string data_type_str = (data_type == DataType::Train) ? "training" : "test";
786 throw std::runtime_error("No " + data_type_str + " data available for R² calculation. Use setInputData() and setOutputData() first.");
787 }
788
789 // Get predictions and targets
790 torch::Tensor predictions;
791 torch::Tensor targets = getTargetData(data_type);
792
793 {
794 torch::NoGradGuard no_grad; // Disable gradients for inference
795 predictions = forward(data_type);
796 }
797
798 // Validate tensor dimensions
799 if (predictions.sizes() != targets.sizes()) {
800 throw std::runtime_error("Prediction and target tensor dimensions don't match.");
801 }
802
803 // Calculate R² = 1 - (SS_res / SS_tot)
804 torch::Tensor target_mean = torch::mean(targets);
805 torch::Tensor ss_res = torch::sum(torch::pow(targets - predictions, 2));
806 torch::Tensor ss_tot = torch::sum(torch::pow(targets - target_mean, 2));
807
808 // Handle edge case where all target values are the same
809 if (ss_tot.item<double>() == 0.0) {
810 return (ss_res.item<double>() == 0.0) ? 1.0 : -std::numeric_limits<double>::infinity();
811 }
812
813 double r_squared = 1.0 - (ss_res.item<double>() / ss_tot.item<double>());
814
815 return r_squared;
816}

862std::map<std::string, double> NeuralNetworkWrapper::evaluate(const torch::Tensor& test_inputs,
863 const torch::Tensor& test_targets) {
864 if (!is_initialized_) {
865 throw std::runtime_error("Network must be initialized before evaluation.");
866 }
867
868 // Validate input tensors
869 if (!test_inputs.defined() || !test_targets.defined()) {
870 throw std::runtime_error("Input tensors must be defined for evaluation.");
871 }
872
873 if (test_inputs.size(0) != test_targets.size(0)) {
874 throw std::runtime_error("Input and target tensors must have the same number of samples.");
875 }
876
877 // Ensure we're in evaluation mode and no gradients are computed
878 this->eval();
879
880 torch::Tensor predictions;
881 {
882 torch::NoGradGuard no_grad; // Critical: disable gradients completely
883 predictions = forward_internal(test_inputs.detach()); // Detach inputs from computation graph
884 }
885
886 // Compute metrics without gradients
887 std::map<std::string, double> metrics;
888
889 // Detach all tensors before computing metrics
890 torch::Tensor pred_detached = predictions.detach();
891 torch::Tensor target_detached = test_targets.detach();
892
893 // Mean Squared Error
894 torch::Tensor mse = torch::mse_loss(pred_detached, target_detached);
895 metrics["mse"] = mse.item<double>();
896
897 // Root Mean Squared Error
898 metrics["rmse"] = std::sqrt(metrics["mse"]);
899
900 // Mean Absolute Error
901 torch::Tensor mae = torch::mean(torch::abs(pred_detached - target_detached));
902 metrics["mae"] = mae.item<double>();
903
904 // R-squared
905 torch::Tensor target_mean = torch::mean(target_detached);
906 torch::Tensor ss_res = torch::sum(torch::pow(target_detached - pred_detached, 2));
907 torch::Tensor ss_tot = torch::sum(torch::pow(target_detached - target_mean, 2));
908 double r_squared = 1.0 - (ss_res.item<double>() / ss_tot.item<double>());
909 metrics["r_squared"] = r_squared;
910
911 metrics["sample_count"] = static_cast<double>(test_inputs.size(0));
912
913 std::cout << "Evaluation completed on " << test_inputs.size(0) << " samples:" << std::endl;
914 std::cout << " MSE: " << std::fixed << std::setprecision(6) << metrics["mse"] << std::endl;
915 std::cout << " RMSE: " << std::fixed << std::setprecision(6) << metrics["rmse"] << std::endl;
916 std::cout << " MAE: " << std::fixed << std::setprecision(6) << metrics["mae"] << std::endl;
917 std::cout << " R²: " << std::fixed << std::setprecision(6) << metrics["r_squared"] << std::endl;
918
919 return metrics;
920}

996void NeuralNetworkWrapper::setInputDataFromHyperParams(DataType data_type,
997 const TimeSeriesSet<double>& time_series_set,
998 double t_start, double t_end, double dt) {
999 if (!is_initialized_) {
1000 throw std::runtime_error("Network must be initialized before setting input data.");
1001 }
1002
1003 const auto& selected_series = hyperparams_.getSelectedSeriesIds();
1004 const auto& all_lags = hyperparams_.getLags();
1005 const auto& lag_multipliers = hyperparams_.getLagMultiplier();
1006
1007 if (selected_series.empty()) {
1008 throw std::runtime_error("No time series selected in hyperparameters.");
1009 }
1010
1011 // Validate that time_series_set has enough series
1012 for (int series_idx : selected_series) {
1013 if (series_idx >= static_cast<int>(time_series_set.size())) {
1014 throw std::runtime_error("Selected series index " + std::to_string(series_idx) +
1015 " exceeds TimeSeriesSet size (" + std::to_string(time_series_set.size()) + ").");
1016 }
1017 }
1018
1019 // Store original series names
1020 setOriginalSeriesNames(time_series_set.getSeriesNames());
1021
1022 // Calculate total features from selected series
1023 int total_features = 0;
1024 for (int series_idx : selected_series) {
1025 total_features += all_lags[series_idx].size();
1026 }
1027
1028 if (total_features != input_size_) {
1029 throw std::runtime_error("Calculated input features (" + std::to_string(total_features) +
1030 ") don't match network input size (" + std::to_string(input_size_) + ").");
1031 }
1032
1033 // Calculate number of time steps
1034 int num_time_steps = static_cast<int>((t_end - t_start) / dt) + 1;
1035
1036 // Create input tensor [num_samples, total_features]
1037 torch::Tensor input_tensor = torch::zeros({num_time_steps, total_features}, torch::kFloat32);
1038
1039 int feature_index = 0;
1040
1041 // Process each selected time series with its lags and multipliers
1042 for (int series_idx : selected_series) {
1043 const auto& current_series = time_series_set[series_idx];
1044 const auto& base_lags = all_lags[series_idx];
1045 int multiplier = lag_multipliers[series_idx];
1046
1047 // For each lag in this series, apply the multiplier
1048 for (int base_lag : base_lags) {
1049 int actual_lag = base_lag * multiplier;
1050
1051 // Fill the feature column for this actual lag
1052 for (int time_idx = 0; time_idx < num_time_steps; ++time_idx) {
1053 double current_time = t_start + time_idx * dt;
1054 double lag_time = current_time - actual_lag * dt; // Apply actual lag
1055
1056 // Interpolate value at lagged time
1057 double lagged_value = current_series.interpol(lag_time);
1058 input_tensor[time_idx][feature_index] = static_cast<float>(lagged_value);
1059 }
1060 feature_index++;
1061 }
1062 }
1063
1064 // Store the processed input data
1065 if (data_type == DataType::Train) {
1066 train_input_data_ = input_tensor;
1067 } else {
1068 test_input_data_ = input_tensor;
1069 }
1070
1071 std::cout << "Input data created: " << num_time_steps << " samples, "
1072 << total_features << " features" << std::endl;
1073}
