155// Function to randomly select an Individual based on inverse rank probability
156template<class T>
157const Individual& GeneticAlgorithm<T>::selectIndividualByRank() {
158 // Calculate weights as the inverse of rank
159 std::vector<double> weights(Individuals.size());
160 for (size_t i = 0; i < Individuals.size(); ++i) {
161 weights[i] = 1.0 / Individuals[i].rank;
162 }
163
164 // Create a cumulative probability distribution
165 std::vector<double> cumulative(weights.size());
166 std::partial_sum(weights.begin(), weights.end(), cumulative.begin());
167
168 // Normalize the cumulative probabilities
169 double totalWeight = cumulative.back();
170 for (double& value : cumulative) {
171 value /= totalWeight;
172 }
173
174 // Generate a random number between 0 and 1
175 std::random_device rd;
176 std::mt19937 gen(rd());
177 std::uniform_real_distribution<> dis(0.0, 1.0);
178 double randomValue = dis(gen);
179
180 // Find the corresponding individual
181 for (size_t i = 0; i < cumulative.size(); ++i) {
182 if (randomValue <= cumulative[i]) {
183 return Individuals[i];
184 }
185 }
186
187 // Fallback (shouldn't be reached)
188 return Individuals.back();
189}

191void SortIndices(const std::vector<Individual>& individuals, std::vector<int>& indices) {
192 size_t n = indices.size();
193
194 // Perform Bubble Sort on indices based on fitness values
195 for (size_t i = 0; i < n - 1; ++i) {
196 for (size_t j = 0; j < n - i - 1; ++j) {
197 // Compare fitness values of the indices
198 if (individuals[indices[j]].fitness > individuals[indices[j + 1]].fitness) {
199 // Swap indices
200 std::swap(indices[j], indices[j + 1]);
201 }
202 }
203 }
